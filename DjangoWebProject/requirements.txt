# Install Oracle instant client for accessing databese
# ==========================================================================
# Instruction for Windows
# --------------------------------------------------------------------------
# Download the basic package from the following link
# https://www.oracle.com/database/technologies/instant-client/downloads.html
# Unzip and place in the following directory
# C:\oracle\
# Place the files from the google drive link in slack to
# C:\oracle\instantclient_xx_x\network\admin (_xx_x is the version #)
# Use the search function and type environment variables
# Click Environment Variables -> Click System variables Path -> Edit -> New
# Past the following
# C:\oracle\instantclient_xx_x (_xx_x is the version #)
# Click ok

#  --------------------------------------------------------------------------
# Instruction for MacOS
# --------------------------------------------------------------------------
# Download the basic package from the following link
# https://www.oracle.com/database/technologies/instant-client/downloads.html
# Unzip and place in the following directory
# MacHDD -> Users -> Current Users
# Place the files from the google drive link in slack to
# ~/instantclient_xx_x/network/admin (_xx_x is the version #)
# Use the terminal
# export PATH=~/instantclient_xx_x:$PATH (_xx_x is the version #)
